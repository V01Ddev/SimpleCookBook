\chapter{Desserts}

\section{Hot Chocolate}
\subsubsection{Ingredients}
For one cup
\begin{enumerate}
    \item 1 cup of Milk
    \item 1 tsp of cocoa powder
    \item 1 tsp of sugar
    \item 1 tsp of chocolate, nutella or otherwise
\end{enumerate}

\subsubsection{Cooking}
For one serving:
\begin{enumerate}
    \item Add ingredients
    \item Wait for it to reach a light boil
    \item Serve
\end{enumerate}
